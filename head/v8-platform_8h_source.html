472// Opaque type representing a handle to a shared memory region.
473class SharedMemoryHandle {
474 public:
475 // For the handle itself, we use the underlying type (e.g. unsigned int)
476 // instead of e.g. mach_port_t to avoid pulling in large OS header files into
477 // this header file. Instead, the users of these routines are expected to
478 // include the respective OS headers in addition to this one.
479
480#if V8_OS_DARWIN
481 // A mach_port_t referencing a memory entry object.
482 using PlatformHandle = unsigned int;
483#elif V8_OS_FUCHSIA
484 // A zx_handle_t to a VMO.
485 using PlatformHandle = uint32_t;
486#elif V8_OS_WIN
487 // A Windows HANDLE to a file mapping object.
488 using PlatformHandle = void*;
489#else
490 // A file descriptor.
491 using PlatformHandle = int;
492#endif
493
494 static constexpr SharedMemoryHandle FromPlatformHandle(
495 PlatformHandle handle) {
496 return SharedMemoryHandle(handle);
497 }
498
499 PlatformHandle GetPlatformHandle() const { return handle_; }
500
501 private:
502 SharedMemoryHandle() = delete;
503 explicit constexpr SharedMemoryHandle(PlatformHandle handle)
504 : handle_(handle) {}
505
506 PlatformHandle handle_;
507};

1072 * Allocate a subspace.
1073 *
1074 * The address space of a subspace stays reserved in the parent space for the
1075 * lifetime of the subspace. As such, it is guaranteed that page allocations
1076 * on the parent space cannot end up inside a subspace.
1077 *
1078 * \param hint Hints where the subspace should be allocated. See
1079 * AllocatePages() for more details.
1080 *
1081 * \param size The size in bytes of the subspace. Must be a multiple of the
1082 * allocation_granularity().
1083 *
1084 * \param alignment The alignment of the subspace in bytes. Must be a multiple
1085 * of the allocation_granularity() and should be a power of two.
1086 *
1087 * \param max_page_permissions The maximum permissions that pages allocated in
1088 * the subspace can obtain.
1089 *
1090 * \param key Optional memory protection key for the subspace. If used, the
1091 * returned subspace will use this key for all its memory pages.
1092 *
1093 * \param handle Optional file descriptor for the subspace. If used, the
1094 * returned subspace will use this file descriptor with 0 offset as the
1095 * space's underlying file.
1096 *
1097 * \returns a new subspace or nullptr on failure.
1098 */
