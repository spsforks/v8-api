462Local<Value> Context::GetEmbedderData(int index) {
463#ifndef V8_ENABLE_CHECKS
464 using A = internal::Address;
465 using I = internal::Internals;
466 A ctx = internal::ValueHelper::ValueAsAddress(this);
467 A embedder_data =
468 I::ReadTaggedPointerField(ctx, I::kNativeContextEmbedderDataOffset);
469 int value_offset =
470 I::kEmbedderDataArrayHeaderSize + (I::kEmbedderDataSlotSize * index);
471 A value = I::ReadRawField<A>(embedder_data, value_offset);
472#ifdef V8_COMPRESS_POINTERS
473 // We read the full pointer value and then decompress it in order to avoid
474 // dealing with potential endiannes issues.
475 value = I::DecompressTaggedField(embedder_data, static_cast<uint32_t>(value));
476#endif
477
478 auto* isolate = I::GetCurrentIsolate();
479 return Local<Value>::New(isolate, value);
480#else
481 return SlowGetEmbedderData(index);
482#endif
483}

485void* Context::GetAlignedPointerFromEmbedderData(Isolate* isolate, int index,
486 EmbedderDataTypeTag tag) {
487#if !defined(V8_ENABLE_CHECKS)
488 using A = internal::Address;
489 using I = internal::Internals;
490 A ctx = internal::ValueHelper::ValueAsAddress(this);
491 A embedder_data =
492 I::ReadTaggedPointerField(ctx, I::kNativeContextEmbedderDataOffset);
493 int value_offset = I::kEmbedderDataArrayHeaderSize +
494 (I::kEmbedderDataSlotSize * index) +
495 I::kEmbedderDataSlotExternalPointerOffset;
496 return reinterpret_cast<void*>(I::ReadExternalPointerField(
497 isolate, embedder_data, value_offset, ToExternalPointerTag(tag)));
498#else
499 return SlowGetAlignedPointerFromEmbedderData(index, tag);
500#endif
501}

503void* Context::GetAlignedPointerFromEmbedderData(int index,
504 EmbedderDataTypeTag tag) {
505#if !defined(V8_ENABLE_CHECKS)
506 using A = internal::Address;
507 using I = internal::Internals;
508 A ctx = internal::ValueHelper::ValueAsAddress(this);
509 A embedder_data =
510 I::ReadTaggedPointerField(ctx, I::kNativeContextEmbedderDataOffset);
511 int value_offset = I::kEmbedderDataArrayHeaderSize +
512 (I::kEmbedderDataSlotSize * index) +
513 I::kEmbedderDataSlotExternalPointerOffset;
514 Isolate* isolate = I::GetCurrentIsolateForSandbox();
515 return reinterpret_cast<void*>(I::ReadExternalPointerField(
516 isolate, embedder_data, value_offset, ToExternalPointerTag(tag)));
517#else
518 return SlowGetAlignedPointerFromEmbedderData(index, tag);
519#endif

522template <class T>
523MaybeLocal<T> Context::GetDataFromSnapshotOnce(size_t index) {
524 if (auto repr = GetDataFromSnapshotOnce(index);
525 repr != internal::ValueHelper::kEmpty) {
526 internal::PerformCastCheck(internal::ValueHelper::ReprAsValue<T>(repr));
527 return Local<T>::FromRepr(repr);
528 }
529 return {};
530}
