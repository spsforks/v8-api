956Local<Data> Object::GetInternalField(int index) {
957#ifndef V8_ENABLE_CHECKS
958 using A = internal::Address;
959 using I = internal::Internals;
960 A obj = internal::ValueHelper::ValueAsAddress(this);
961 // Fast path: If the object is a plain JSObject, which is the common case, we
962 // know where to find the internal fields and can return the value directly.
963 int instance_type = I::GetInstanceType(obj);
964 if (I::CanHaveInternalField(instance_type)) {
965 int offset = I::kJSAPIObjectWithEmbedderSlotsHeaderSize +
966 (I::kEmbedderDataSlotSize * index);
967 A value = I::ReadRawField<A>(obj, offset);
968#ifdef V8_COMPRESS_POINTERS
969 // We read the full pointer value and then decompress it in order to avoid
970 // dealing with potential endianness issues.
971 value = I::DecompressTaggedField(obj, static_cast<uint32_t>(value));
972#endif
973
974 auto* isolate = I::GetCurrentIsolate();
975 return Local<Data>::New(isolate, value);
976 }
977#endif
978 return SlowGetInternalField(index);
979}

981void* Object::GetAlignedPointerFromInternalField(v8::Isolate* isolate,
982 int index,
983 EmbedderDataTypeTag tag) {
984#if !defined(V8_ENABLE_CHECKS)
985 using A = internal::Address;
986 using I = internal::Internals;
987 A obj = internal::ValueHelper::ValueAsAddress(this);
988 // Fast path: If the object is a plain JSObject, which is the common case, we
989 // know where to find the internal fields and can return the value directly.
990 auto instance_type = I::GetInstanceType(obj);
991 if (V8_LIKELY(I::CanHaveInternalField(instance_type))) {
992 int offset = I::kJSAPIObjectWithEmbedderSlotsHeaderSize +
993 (I::kEmbedderDataSlotSize * index) +
994 I::kEmbedderDataSlotExternalPointerOffset;
995 A value = I::ReadExternalPointerField(isolate, obj, offset,
996 ToExternalPointerTag(tag));
997 return reinterpret_cast<void*>(value);
998 }
999#endif
1000 return SlowGetAlignedPointerFromInternalField(isolate, index, tag);
1001}

1027T* Object::Unwrap(v8::Isolate* isolate, const v8::Local<v8::Object>& wrapper) {
1028 CppHeapPointerTagRange tag_range(tag, tag);
1029 auto obj = internal::ValueHelper::ValueAsAddress(*wrapper);
1030#if !defined(V8_ENABLE_CHECKS)
1031 return internal::ReadCppHeapPointerField<T>(
1032 isolate, obj, internal::Internals::kJSObjectHeaderSize, tag_range);
1033#else // defined(V8_ENABLE_CHECKS)
1034 return reinterpret_cast<T*>(Unwrap(isolate, obj, tag_range));
1035#endif // defined(V8_ENABLE_CHECKS)
1036}
